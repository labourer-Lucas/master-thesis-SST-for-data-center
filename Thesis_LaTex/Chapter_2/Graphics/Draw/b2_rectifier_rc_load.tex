\begin{tikzpicture}[circuit ee IEC, set diode graphic=var diode IEC graphic, tiny circuit symbols]
%Diodes
\node[diode, small circuit symbols, rotate = 90, name = d1] at (0, 0) {} ++(1, 0) node[diode, small circuit symbols, rotate = 90, name = d2] {} ++(0, -2.5) node[diode, small circuit symbols, rotate = 90, name = d4] {} ++ (-1, 0) node[diode, small circuit symbols, rotate = 90, name = d3] {};

%Connections of the diodes
\draw[-] (d3) -- (d1) node[contact, pos=0.8, name = pos_contact] {};
\draw[-] (d4) -- (d2) node[contact, pos=0.2, name = neg_contact] {};

%Input terminals
\draw[-] (pos_contact) -- ++(-1.5, 0) node[draw, circle, inner sep = 1.25pt, fill = white] (in_plus) {};
\draw[-] (neg_contact) -- ++(-2.5, 0) node[draw, circle, inner sep = 1.25pt, fill = white] (in_minus) {};

%Input current
\draw[-triangle 45] (in_plus) -- +(0.85, 0);
\node[above left] at ($(in_plus) + (1, 0.05)$) {$i_\text{l}$};

%Input voltage
\draw[-triangle 45] ($(in_plus) + (0, -0.15)$) -- ($(in_minus) + (0, 0.15)$) node[midway, left] {$v_\text{l}$};

%Diode contacts
\draw[-] (d1) -- ++(0, 0.6) -| (d2);
\draw[-] (d3) -- ++(0, -0.6) -| (d4);
\node[contact, name = d2_contact] at ($(d2) + (0, 0.6)$) {};
\node[contact, name = d4_contact] at ($(d4) + (0, -0.6)$) {};

%Capacitor
\draw[-] (d2_contact) -- ++(1, 0) node[contact] (c_plus) {};
\draw[-] (d4_contact) -- ++(1, 0) node[contact] (c_minus) {};
\path (c_plus) -- (c_minus) node[midway, capacitor={info=$C_\text{l}$}, small circuit symbols, rotate = 90, name = C] {};
\draw[-] (c_plus) -- (C.east) (C.west) -- (c_minus);

%Resistor
\draw[-] (c_plus) -- ++(1.25, 0) |- (c_minus);
\path ($(c_plus) + (1.25, 0)$) -- ($(c_minus) + (1.25, 0)$) node[resistor={info=$R_\text{l}$}, midway, small circuit symbols, rotate = 90, name = R, circuit symbol size = width 3.5 height 1.5, fill = white] {};
\draw[-triangle 45] ($(R.east) + (0.4, 0.2)$) -- ($(R.west) + (0.4, -0.2)$) node[midway, right] {$v_\text{r}$};
\end{tikzpicture}
